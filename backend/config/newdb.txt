package config

import (
	"fmt"
	"log"
	"os"
	"strings"
	"time"

	"github.com/Tawunchai/openvas/entity"

	"gorm.io/driver/postgres"
	"gorm.io/gorm"
)

var db *gorm.DB

func DB() *gorm.DB {
	return db
}

func ConnectDB() {
	// 1) ถ้ามี env ให้ใช้ก่อน
	envDSN := strings.TrimSpace(os.Getenv("DATABASE_URL"))
	if envDSN != "" {
		tryConnectWithRetry(envDSN, "DATABASE_URL")
		return
	}

	// 2) ไม่มี envDSN -> ลองแบบรันบนเครื่อง (host=localhost port=15432 ตาม compose ports)
	localDSN := "host=localhost port=15432 user=pbi password=Pbi12345 dbname=gvmd sslmode=disable"
	if tryConnectOnce(localDSN) == nil {
		tryConnectWithRetry(localDSN, "fallback-localhost:15432")
		return
	}

	// 3) ถ้า localhost ไม่ได้ -> ลองแบบรันใน docker network (service name)
	dockerDSN := "host=pg-gvm port=5432 user=pbi password=Pbi12345 dbname=gvmd sslmode=disable"
	tryConnectWithRetry(dockerDSN, "fallback-docker(pg-gvm:5432)")
}

func tryConnectWithRetry(dsn string, label string) {
	maxRetries := 20
	retryDelay := 2 * time.Second

	var database *gorm.DB
	var err error

	log.Printf("🔌 Connecting DB using: %s\n", label)

	for i := 1; i <= maxRetries; i++ {
		database, err = gorm.Open(postgres.Open(dsn), &gorm.Config{})
		if err == nil {
			sqlDB, err2 := database.DB()
			if err2 == nil {
				sqlDB.SetMaxIdleConns(5)
				sqlDB.SetMaxOpenConns(20)
				sqlDB.SetConnMaxLifetime(30 * time.Minute)

				if pingErr := sqlDB.Ping(); pingErr == nil {
					db = database
					fmt.Println("✅ Database connected successfully")
					return
				} else {
					err = pingErr
				}
			} else {
				err = err2
			}
		}

		log.Printf("⏳ DB not ready (attempt %d/%d): %v\n", i, maxRetries, err)
		time.Sleep(retryDelay)
	}

	log.Fatalf("❌ failed to connect to PostgreSQL after %d attempts: %v", maxRetries, err)
}

func tryConnectOnce(dsn string) error {
	database, err := gorm.Open(postgres.Open(dsn), &gorm.Config{})
	if err != nil {
		return err
	}
	sqlDB, err2 := database.DB()
	if err2 != nil {
		return err2
	}
	return sqlDB.Ping()
}

// ✅ สร้างตารางจาก Entity
func SetupDatabase() {
	if db == nil {
		log.Fatal("❌ database is nil, call ConnectDB() before SetupDatabase()")
	}

	err := db.AutoMigrate(
		&entity.AppGroup{},
		&entity.AppLineMaster{},
		&entity.Notification{},
	)
	if err != nil {
		log.Fatalf("❌ AutoMigrate failed: %v", err)
	}

	fmt.Println("✅ AutoMigrate completed (AppGroup, AppLineMaster, Notification)")
}
